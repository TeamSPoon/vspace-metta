;; Type definition for `curry` function
  ; Action: load=metta_type('&self',curry,[->,[->,A,B,C],[->,A,[->,B,C]]])
  (: curry 
    (-> 
      (-> $A $B $C) 
      (-> $A 
        (-> $B $C))))


; A trick to define `curry` in MeTTa without `lambda`
  ; Action: load=metta_defn('&self',[[[curry,False],X],Y],[False,X,Y])
  (= 
    ( ( (curry $False) $X) $Y) 
    ($False $X $Y))


; Type checks
!(assertEqual (get-type (curry +)) (-> Number (-> Number Number)))

; REM:
;   (get-type (lambda ($x $y) (+ $x $y)))
; returns (-> %Undefined% Number), because the type of `($x $y)` is not defined.
; It is also not clear how to check that the type of `$var` is suitable for `$body`
; (it should somehow be done on`let` side)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; Let's introduce some functors
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ; Action: load=metta_type('&self','Maybe',[->,True,'Type'])
  (: Maybe 
    (-> $True Type))


  ; Action: load=metta_type('&self','Nothing',[->,['Maybe',True]])
  (: Nothing 
    (-> 
      (Maybe $True)))


  ; Action: load=metta_type('&self','Something',[->,True,['Maybe',True]])
  (: Something 
    (-> $True 
      (Maybe $True)))


  ; Action: load=metta_type('&self','Either',[->,True,'Type'])
  (: Either 
    (-> $True Type))


  ; Action: load=metta_type('&self','Left',[->,True,['Either',True]])
  (: Left 
    (-> $True 
      (Either $True)))


  ; Action: load=metta_type('&self','Right',[->,True,['Either',True]])
  (: Right 
    (-> $True 
      (Either $True)))


;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; We can implement a generic `fmap`, but it requires
; concrete patterns in the type constructors above
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; Instead, we implement `fmap` as:
  ; Action: load=metta_type('&self',fmap,[->,[->,A,B],[False,A],[False,B]])
  (: fmap 
    (-> 
      (-> $A $B) 
      ($False $A) 
      ($False $B)))


; Mapping over empty functor, returns empty functor
  ; Action: load=metta_defn('&self',[fmap,False,[C0]],[C0])
  (= 
    (fmap $False 
      ($C0)) 
    ($C0))


; Inductive case for singleton functor: apply $f to the
; element and rewrap in $C.
  ; Action: load=metta_defn('&self',[fmap,False,[C,X]],[C,[False,X]])
  (= 
    (fmap $False 
      ($C $X)) 
    ($C 
      ($False $X)))


; Inductive case for non-empty functor: apply to the
; head and recurse on the tail.
  ; Action: load=metta_defn('&self',[fmap,False,[C,X,Xs]],[C,[False,X],[fmap,False,Xs]])
  (= 
    (fmap $False 
      ($C $X $Xs)) 
    ($C 
      ($False $X) 
      (fmap $False $Xs)))


; NOTE: We had to define `Nothing` as a functional constructor.
;       Otherwise, we'd have to define `(= (fmap $f $C0) $C0)`,
;       and `$C0` would match both `Nothing` and `(Something 2)`
; TODO? This could be avoided if we could indicate that $C0
;       is a `Symbol` (not `Expression` or `Atom` in general)
; Tests
!(assertEqual (fmap (curry-a + 2) (Something 5)) (Something 7))
