;; `match` searches for all expressions corresponding to
;; the given pattern and produces the output pattern.
;; It doesn''t search in subexpressions.

;; Some expressions to be matched
%~ [ ( convert  =
%~     [ leaf1(leaf2),
%~       leaf3] =
%~     'True'),
%~   join_into_body( leaf1(leaf2,Leaf1_Leaf2),[Leaf1_Leaf2,leaf3],
%~     true,
%~     leaf1(leaf2,_416020),
%~     [_416020,leaf3])]
%~ [convert=([leaf1(leaf2),leaf3]='True'),join_into_body((leaf1(leaf2,_416020),[_416020,leaf3]),true,leaf1(leaf2,_416020),[_416020,leaf3])]
  ; Action: load=metta_atom('&self',[[leaf1,leaf2],leaf3])
  ( (leaf1 leaf2) leaf3)
 Compiled(&self): leaf1(leaf2, A) :-
    [A, leaf3].



%~ [ ( convert  =
%~     [ [ leaf0(leaf1),
%~         leaf2],
%~       leaf3] =
%~     'True'),
%~   join_into_body(
%~      leaf0(leaf1,Leaf0_Leaf1),[[Leaf0_Leaf1,leaf2],leaf3], true,leaf0(leaf1,_35668),
%~      [ [_136638,leaf2],
%~        leaf3])]
%~ [convert=([[leaf0(leaf1),leaf2],leaf3]='True'),join_into_body((leaf0(leaf1,_35668),[[_35668,leaf2],leaf3]),true,leaf0(leaf1,_35668),[[_35668,leaf2],leaf3])]

; `assertEqualToResult` checks all the results of the first
; expression; it doesn''t evaluate the second expression,
; which is treated as a set of expected results.
!(assertEqualToResult (match &self (($X leaf2) leaf3) $X) (leaf1 (leaf0 leaf1)))

exec(A) :-
    leaf0(leaf1, B),
    leaf1(B, C),
    assertEqualToResult(match('&self', [[X, leaf2], leaf3], X),
                        C,
                        A),
    true.
