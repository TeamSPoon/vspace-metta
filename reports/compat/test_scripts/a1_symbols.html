;; `match` searches for all expressions corresponding to
;; the given pattern and produces the output pattern.
;; It doesn''t search in subexpressions.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Some expressions to be matched
  ; Action: load=metta_atom('&self',[[leaf1,leaf2],leaf3])
  ( (leaf1 leaf2) leaf3)


  ; Action: load=metta_atom('&self',[[[leaf0,leaf1],leaf2],leaf3])
  ( ( (leaf0 leaf1) leaf2) leaf3)


; This one contains `((leaf1 leaf2) leaf3)` as a subexpression
; and thus will not be matched
  ; Action: load=metta_atom('&self',[top,[[leaf1,leaf2],leaf3]])
  (top 
    ( (leaf1 leaf2) leaf3))


; `assertEqualToResult` checks all the results of the first
; expression; it doesn''t evaluate the second expression,
; which is treated as a set of expected results.
!(assertEqualToResult (match &self (($X leaf2) leaf3) $X) (leaf1 (leaf0 leaf1)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; Symbols can be arranged in arbitrary expressions
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ; Action: load=metta_atom('&self',['Sam',is,a,frog])
  (Sam is a frog)


  ; Action: load=metta_atom('&self',['Tom',is,a,cat])
  (Tom is a cat)


  ; Action: load=metta_atom('&self',['Sophia',is,a,robot])
  (Sophia is a robot)


; `match` can bind more that one variable
!(assertEqualToResult (match &self ($Who is a $What) ($Who the $What)) ((Sam the frog) (Tom the cat) (Sophia the robot)))


;; TEST-SCRIPTS.A1-SYMBOLS.02

(@ loonit_success 
  (=@=  
    ( (Sam the frog) 
      (Sophia the robot) 
      (Tom the cat)) 
    ( (Sam the frog) 
      (Sophia the robot) 
      (Tom the cat))))
 True

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; More examples of pattern matching
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; `:=` is a custom symbol. These are still purely symbolic expressions
  ; Action: load=metta_atom('&self',[:=,['Green','Sam'],'True'])
  (:= 
    (Green Sam) True)


  ; Action: load=metta_atom('&self',[:=,['White','Tom'],'True'])
  (:= 
    (White Tom) True)


  ; Action: load=metta_atom('&self',[:=,['Green','Tom'],'False'])
  (:= 
    (Green Tom) False)


!(assertEqualToResult (match &self (:= (Green $Who) True) ($Who is really green)) ((Sam is really green)))


;; TEST-SCRIPTS.A1-SYMBOLS.03

(@ loonit_success 
  (=@=  
    ( (Sam is really green)) 
    ( (Sam is really green))))
 True

!(assertEqualToResult (match &self (:= ($Color $Who) True) ($Who is really $Color)) ((Sam is really Green) (Tom is really White)))
