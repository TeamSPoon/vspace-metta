; The result of matching is also chained
; Example from OpenCog Classic wiki on PLN Backward Chaining
; `And` and `T` are custom symbols (they are used here
; to avoid mixing them up with symbols from common lib)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; Some facts in the knowledge base
  ; Action: load=metta_atom('&self',['Evaluation',[philosopher,'Plato']])
  (Evaluation 
    (philosopher Plato))


  ; Action: load=metta_atom('&self',['Evaluation',['likes-to-wrestle','Plato']])
  (Evaluation 
    (likes-to-wrestle Plato))


; An implication rule
  ; Action: load=metta_atom('&self',['Implication',['And',['Evaluation',[philosopher,X]],['Evaluation',['likes-to-wrestle',X]]],['Evaluation',[human,X]]])
  (Implication 
    (And 
      (Evaluation 
        (philosopher $X)) 
      (Evaluation 
        (likes-to-wrestle $X))) 
    (Evaluation 
      (human $X)))


; Another implication rule
  ; Action: load=metta_atom('&self',['Implication',['Evaluation',[human,X]],['Evaluation',[mortal,X]]])
  (Implication 
    (Evaluation 
      (human $X)) 
    (Evaluation 
      (mortal $X)))


; Deduction case when the desired evaluation is present in
; the knowledge base
  ; Action: load=metta_defn('&self',[deduce,['Evaluation',[P,X]]],[match,'&self',['Evaluation',[P,X]],'True'])
  (= 
    (deduce 
      (Evaluation 
        ($P $X))) 
    (match &self 
      (Evaluation 
        ($P $X)) True))


; Deduction case when the desired evaluation is the result
; of an implication, which implies a recursion
  ; Action: load=metta_defn('&self',[deduce,['Evaluation',[P,X]]],[match,'&self',['Implication',A,['Evaluation',[P,X]]],[deduce,A]])
  (= 
    (deduce 
      (Evaluation 
        ($P $X))) 
    (match &self 
      (Implication $A 
        (Evaluation 
          ($P $X))) 
      (deduce $A)))


; Deduction case for generic "And" expressions;
; also recursive
  ; Action: load=metta_defn('&self',[deduce,['And',A,B]],['And',[deduce,A],[deduce,B]])
  (= 
    (deduce 
      (And $A $B)) 
    (And 
      (deduce $A) 
      (deduce $B)))


; True & True = True
  ; Action: load=metta_defn('&self',['And','True','True'],'True')
  (= 
    (And True True) True)


; Test deduction that Plato is mortal
!(assertEqual (deduce (Evaluation (mortal Plato))) True)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; Another example of backchaining
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; Definition of explain when the desired Evaluation is
; in the knowledge base
  ; Action: load=metta_defn('&self',[explain,['Evaluation',[P,X]]],[match,'&self',['Evaluation',[P,X]],[P,X]])
  (= 
    (explain 
      (Evaluation 
        ($P $X))) 
    (match &self 
      (Evaluation 
        ($P $X)) 
      ($P $X)))


; Definition of explain when the desired Evaluation is
; the result of an implication
  ; Action: load=metta_defn('&self',[explain,['Evaluation',[P,X]]],[match,'&self',['Implication',A,['Evaluation',[P,X]]],[[P,X],proven,by,[explain,A]]])
  (= 
    (explain 
      (Evaluation 
        ($P $X))) 
    (match &self 
      (Implication $A 
        (Evaluation 
          ($P $X))) 
      ( ($P $X) proven by 
        (explain $A))))


; Definition of explain for And
  ; Action: load=metta_defn('&self',[explain,['And',A,B]],['And',[explain,A],[explain,B]])
  (= 
    (explain 
      (And $A $B)) 
    (And 
      (explain $A) 
      (explain $B)))


; Example of explain why Plato is mortal
!(assertEqual (explain (Evaluation (mortal Plato))) ((mortal Plato) proven by ((human Plato) proven by (And (philosopher Plato) (likes-to-wrestle Plato)))))
