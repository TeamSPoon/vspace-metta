;; used for querying and reasoning over declarative knowledge

;; Fact
%~ [ convert=('Frog'('Sam')='True'),
%~   join_into_body(('Frog'('Sam','True'),true),true,'Frog'('Sam','True'),true)]
%~ [convert=('Frog'('Sam')='True'),join_into_body(('Frog'('Sam','True'),true),true,'Frog'('Sam','True'),true)]
  ; Action: load=metta_atom('&self',['Frog','Sam'])
  (Frog Sam)
 Compiled(&self): 'Frog'('Sam', 'True').



%~ [ ( convert  =
%~     frog(X) =
%~     match('&self','Frog'(X),'True')),
%~   join_into_body( frog(X,Frog_X),true,
%~     'Frog'(X,Frog_Self_X),match('&self',Frog_Self_X,'True'),
%~     frog(X,_2544),
%~     'Frog'(X,_2568),match('&self',_2568,'True'))]
%~ [convert=(frog(X)=match('&self','Frog'(X),'True')),join_into_body((frog(X,_2544),true),('Frog'(X,_2568),match('&self',_2568,'True')),frog(X,_2544),('Frog'(X,_2568),match('&self',_2568,'True')))]
  ; Action: load=metta_defn('&self',[frog,X],[match,'&self',['Frog',X],'True'])
  (= 
    (frog $X) 
    (match &self 
      (Frog $X) True))
 Compiled(&self): frog(X, HeadRes) :-
    'Frog'(X, A),
    match('&self', A, 'True').



; `(Frog Sam)` is not reduced; it is just a declaration
!(assertEqualToResult (Frog Sam) ((Frog Sam)))

exec(A) :-
    'Frog'('Sam', B),
    assertEqualToResult(B, [B], A),
    true.
