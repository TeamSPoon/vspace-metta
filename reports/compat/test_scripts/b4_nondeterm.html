;; Nondeterminism in matching and interpretation

  ; Action: load=metta_defn('&self',[color],yellow)

  ; Action: load=metta_defn('&self',[color],red)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; In `a1_symbols` we saw that `match` can return multiple or no results
; The same is true for equality queries
; `superpose` turns a tuple into a non-deterministic result (for comparison here)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
!(assertEqual (match &self (= (color) $X) $X) (superpose (green yellow red)))

(@ loonit_success 
  (equal_enough  
    (green red yellow) 
    (green red yellow)))
        -->(0,'&self',[match,'&self',[=,[color],_275386],_275386],depth(2))
        <--(0,retval(red))
        -->(1,'&self',[superpose,[green,yellow,red]],depth(2))
        <--(1,retval(red))
; Evaluation took 1.71 ms.

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; For the ordinary symbolic expression, if its subexpressions
; are nondeterministic, and no reduction rule is available
; for some combinations of their values, the whole expression
; for these values will appear in the set of final results,
; but without reduction. However, `match` can be called
; directly to filter out non-reducible results
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ; Action: load=metta_defn('&self',[foo],red)
  (= 
    (foo) red)


  ; Action: load=metta_defn('&self',[foo],boo)
  (= 
    (foo) boo)


  ; Action: load=metta_defn('&self',[eq,X,X],'True')
  (= 
    (eq $X $X) True)


  ; Action: load=metta_defn('&self',['find-equal',X,Y],[match,'&self',[=,[eq,X,Y],'True'],X])
  (= 
    (find-equal $X $Y) 
    (match &self 
      (= 
        (eq $X $Y) True) $X))
