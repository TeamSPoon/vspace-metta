;; Example from OpenCog Classic wiki on PLN Backward Chaining
;; No explicit backward chaining is needed
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; facts
  ; Action: load=metta_defn('&self',[croaks,'Fritz'],'True')
  (= 
    (croaks Fritz) True)


  ; Action: load=metta_defn('&self',[eat_flies,'Fritz'],'True')
  (= 
    (eat_flies Fritz) True)


; Rules
  ; Action: load=metta_defn('&self',['And','True','True'],'True')
  (= 
    (And True True) True)


  ; Action: load=metta_defn('&self',[frog,X],['And',[croaks,X],[eat_flies,X]])
  (= 
    (frog $X) 
    (And 
      (croaks $X) 
      (eat_flies $X)))


  ; Action: load=metta_defn('&self',[green,X],[frog,X])
  (= 
    (green $X) 
    (frog $X))


; Conclusion from facts and rules
!(assertEqual (green Fritz) True)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; This is not just functional programming,
; because we can do inference
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ; Action: load=metta_defn('&self',[ift,'True',Then],Then)
  (= 
    (ift True $Then) $Then)


!(assertEqual (ift (green $X) $X) Fritz)


;; TEST-SCRIPTS.B3-DIRECT.02

(@ loonit_success 
  (equal_enough  
    (Fritz) 
    (Fritz)))
 True

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; We can also use expressions with `=` in `match`
; NOTE: `=` is not an ordinary symbol, and declarative reasoning over expressions
; with it may work differently from that over purely symbolic expressions
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
!(assertEqualToResult (match &self (= ($P Fritz) True) $P) (croaks eat_flies))


;; TEST-SCRIPTS.B3-DIRECT.03

(@ loonit_success 
  (=@=  
    (croaks eat_flies) 
    (croaks eat_flies)))
