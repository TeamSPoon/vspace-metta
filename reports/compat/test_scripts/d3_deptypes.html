;; Simple type definitions:
;; Defines the natural numbers `Nat` for counting
  ; Action: load=metta_atom('&self',[:,'Nat','Type'])
  (: Nat Type)


  ; Action: load=metta_atom('&self',[:,'Z','Nat'])
  (: Z Nat)


  ; Action: load=metta_atom('&self',[:,'S',[->,'Nat','Nat']])
  (: S 
    (-> Nat Nat))


; `Vec` takes a `$t` type and a `Nat` length.
  ; Action: load=metta_atom('&self',[:,'Vec',[->,True,'Nat','Type']])
  (: Vec 
    (-> $True Nat Type))


; `Cons` and `Nil` construct `Vec` values, with `Cons` incrementing the length by 1.
  ; Action: load=metta_atom('&self',[:,'Cons',[->,True,['Vec',True,X],['Vec',True,['S',X]]]])
  (: Cons 
    (-> $True 
      (Vec $True $X) 
      (Vec $True 
        (S $X))))


  ; Action: load=metta_atom('&self',[:,[],['Vec',True,'Z']])
  (: () 
    (Vec $True Z))


; We can query the derived type
!(assertEqual (get-type (Cons 0 (Cons 1 ()))) (Vec Number (S (S Z))))

; `drop` removes the head from a `Vec`, returning a `Vec` with length decrease by 1
  ; Action: load=metta_atom('&self',[:,drop,[->,['Vec',True,['S',X]],['Vec',True,X]]])

    (-> 
      (Vec $True 
        (S $X))

  ; Action: load=metta_defn('&self',[drop,['Cons',X,Xs]],Xs)
