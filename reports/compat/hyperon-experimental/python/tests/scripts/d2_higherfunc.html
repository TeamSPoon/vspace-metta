;; REM:
;;   (get-type (lambda ($x $y) (+ $x $y)))
;; returns (-> %Undefined% Number), because the type of `($x $y)` is not defined.
;; It is also not clear how to check that the type of `$var` is suitable for `$body`
;; (it should somehow be done on`let` side)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Let's introduce some functors
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ; Action: load=metta_atom('&self',:,'Maybe',[[->,T,'Type']])
(: Maybe 
  (-> $T Type))


  ; Action: load=metta_atom('&self',:,'Nothing',[[->,['Maybe',T]]])
(: Nothing 
  (-> 
    (Maybe $T)))


  ; Action: load=metta_atom('&self',:,'Something',[[->,T,['Maybe',T]]])
(: Something 
  (-> $T 
    (Maybe $T)))


  ; Action: load=metta_atom('&self',:,'Either',[[->,T,'Type']])
(: Either 
  (-> $T Type))


  ; Action: load=metta_atom('&self',:,'Left',[[->,T,['Either',T]]])
(: Left 
  (-> $T 
    (Either $T)))


  ; Action: load=metta_atom('&self',:,'Right',[[->,T,['Either',T]]])
(: Right 
  (-> $T 
    (Either $T)))


;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; We can implement a generic `fmap`, but it requires
;; concrete patterns in the type constructors above
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Instead, we implement `fmap` as:
  ; Action: load=metta_atom('&self',:,fmap,[[->,[->,A,B],[F,A],[F,B]]])
(: fmap 
  (-> 
    (-> $A $B) 
    ($F $A) 
    ($F $B)))


;; Mapping over empty functor, returns empty functor
  ; Action: load=metta_atom('&self',=,[fmap,F,[C0]],[C0])
(= 
  (fmap $F 
    ($C0)) 
  ($C0))


;; Inductive case for singleton functor: apply $f to the
;; element and rewrap in $C.
  ; Action: load=metta_atom('&self',=,[fmap,F,[C,X]],[C,[F,X]])
(= 
  (fmap $F 
    ($C $X)) 
  ($C 
    ($F $X)))


;; Inductive case for non-empty functor: apply to the
;; head and recurse on the tail.
  ; Action: load=metta_atom('&self',=,[fmap,F,[C,X,Xs]],[C,[F,X],[fmap,F,Xs]])
(= 
  (fmap $F 
    ($C $X $Xs)) 
  ($C 
    ($F $X) 
    (fmap $F $Xs)))
