;; Pathfinding using a simple graph
;; Graph is represented as facts: (= (edge A B) True) means there''s an one-way edge from A to B.

  (= 
    (edge b e) True)


  ; Action: load=metta_defn('&self',[edge,e,f],'True')
  (= 
    (edge e f) True)


; Pathfinding rule
  ; Action: load=metta_defn('&self',[path,X,X],'True')
  (= 
    (path $X $X) True)


; base case, node to itself
  ; Action: load=metta_defn('&self',[path,X,Y],[and,[edge,X,Z],[path,Z,Y]])
  (= 
    (path $X $Y) 
    (and 
      (edge $X $Z) 
      (path $Z $Y)))
