;; `TV` "metarule" based on `match` to run
;; `.tv` is a custom symbol, which is used to connect
;; PLN-like expressions with their truth values.
;; Some `.tv` facts will be defined below
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ; Action: load=metta_atom('&self',=,['TV',X],[match,'&self',['.tv',X,Stv],Stv])
(= 
  (TV $X) 
  (match &self 
    (.tv $X $Stv) $Stv))


;; Definition of TV for a conjunction
;; we could use `(.tv (And $a $b) (stv ...))` instead of this
  ; Action: load=metta_atom('&self',=,['TV',['And',A,B]],[stv,[min,['s-tv',['TV',A]],['s-tv',['TV',B]]],[min,['c-tv',['TV',A]],['c-tv',['TV',B]]]])
(= 
  (TV 
    (And $A $B)) 
  (stv 
    (min 
      (s-tv 
        (TV $A)) 
      (s-tv 
        (TV $B))) 
    (min 
      (c-tv 
        (TV $A)) 
      (c-tv 
        (TV $B)))))


;; Definition of TV for $x when it's the consequent of an implication
  ; Action: load=metta_atom('&self',=,['TV',X],[match,'&self',['.tv',['Implication',Y,X],[stv,S,C]],[stv,[*,S,['s-tv',['TV',Y]]],[*,C,['c-tv',['TV',Y]]]]])
(= 
  (TV $X) 
  (match &self 
    (.tv 
      (Implication $Y $X) 
      (stv $S $C)) 
    (stv 
      (* $S 
        (s-tv 
          (TV $Y))) 
      (* $C 
        (c-tv 
          (TV $Y))))))
