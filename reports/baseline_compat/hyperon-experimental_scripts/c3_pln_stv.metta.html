;; `TV` "metarule" based on `match` to run
;; `.tv` is a custom symbol, which is used to connect
;; PLN-like expressions with their truth values.
;; Some `.tv` facts will be defined below
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ; Action: load=asserted_metta_atom('&self',[=,['TV','$VAR'(x)],[match,'&self',['.tv','$VAR'(x),'$VAR'(stv)],'$VAR'(stv)]])
(= 
  (TV $x) 
  (match &self 
    (.tv $x $stv) $stv))


;; Definition of TV for a conjunction
;; we could use `(.tv (And $a $b) (stv ...))` instead of this
  ; Action: load=asserted_metta_atom('&self',[=,['TV',['And','$VAR'(a),'$VAR'(b)]],[stv,[min,['s-tv',['TV','$VAR'(a)]],['s-tv',['TV','$VAR'(b)]]],[min,['c-tv',['TV','$VAR'(a)]],['c-tv',['TV','$VAR'(b)]]]]])
(= 
  (TV 
    (And $a $b)) 
  (stv 
    (min 
      (s-tv 
        (TV $a)) 
      (s-tv 
        (TV $b))) 
    (min 
      (c-tv 
        (TV $a)) 
      (c-tv 
        (TV $b)))))


;; Definition of TV for $x when it's the consequent of an implication
  ; Action: load=asserted_metta_atom('&self',[=,['TV','$VAR'(x)],[match,'&self',['.tv',['Implication','$VAR'(y),'$VAR'(x)],[stv,'$VAR'(s),'$VAR'(c)]],[stv,[*,'$VAR'(s),['s-tv',['TV','$VAR'(y)]]],[*,'$VAR'(c),['c-tv',['TV','$VAR'(y)]]]]]])
(= 
  (TV $x) 
  (match &self 
    (.tv 
      (Implication $y $x) 
      (stv $s $c)) 
    (stv 
      (* $s 
        (s-tv 
          (TV $y))) 
      (* $c 
        (c-tv 
          (TV $y))))))
