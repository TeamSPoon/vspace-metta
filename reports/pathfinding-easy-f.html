  ; Action: load=metta_atom('&self',[:,'epath-f',[->,'Symbol','Symbol']])
  (: epath-f

  ; Action: load=metta_defn('&self',['epath-f',X],['epath-f',['edge-f',X]])

    (epath-f

'epath-f'(X, HeadRes) :-
    'edge-f'(X, A),
    'epath-f'(A, HeadRes),
    true.

  ; Action: load=metta_defn('&self',['epath-f',X],X)

    (epath-f $X) $X)
